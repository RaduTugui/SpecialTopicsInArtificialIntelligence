from flask import Flask, request, jsonify
from flask_cors import CORS
from pyngrok import ngrok
import torch, io
from PIL import Image
import base64, threading

app = Flask(__name__)
CORS(app)

# Load your saved model
model.load_state_dict(torch.load('/content/drive/MyDrive/archive/memotion_multimodal.pth',
                                     map_location=device))
model.eval()

@app.route('/predict', methods=['POST'])
def predict():
    data = request.get_json()
    text  = data.get('text', '')
    img_b64 = data.get('image')        # base64 string, optional

    # Process text
    enc = tokenizer(text, max_length=128, truncation=True,
                    padding='max_length', return_tensors='pt')
    with torch.no_grad():
        t = roberta_model(input_ids=enc['input_ids'].to(device),
                          attention_mask=enc['attention_mask'].to(device)
                          ).last_hidden_state[:,0,:]

        if img_b64:
            img = Image.open(io.BytesIO(base64.b64decode(img_b64))).convert('RGB')
            pv  = clip_processor(images=img, return_tensors='pt').pixel_values.to(device)
            i   = clip_model.get_image_features(pixel_values=pv).pooler_output
        else:
            i = torch.zeros(1, 512).to(device)   # text-only fallback

        logits = model(t, i)
        probs  = torch.nn.functional.softmax(logits, dim=1)[0]

    return jsonify({
        'label':      'sarcastic' if probs[1] > probs[0] else 'not_sarcastic',
        'sarcastic':  round(float(probs[1]), 4),
        'not_sarcastic': round(float(probs[0]), 4)
    })

# Start server + tunnel
tunnel = ngrok.connect(5000)
print(f"Public URL: {tunnel.public_url}")   # paste this into the UI
threading.Thread(target=lambda: app.run(port=5000)).start()
